4. Algorithme avec Tant Que
algorithme SommeEntiers;
variables
    n, somme, compteur : entier;
debut
    n := 10;
    somme := 0;
    compteur := 1;
    tant_que compteur <= n faire
        somme := somme + compteur;
        compteur := compteur + 1;
    fin_tant_que;
    ecrire("Somme des ", n, " premiers entiers = ", somme);
fin
5. Algorithme avec Selon/Cas (Switch)
algorithme Calculatrice;

    choix, a, b, resultat : entier;
debut
    a := 15;
    b := 3;
    choix := 2;  #exemple commentaire
    selon (choix)
        si 1 : 
            resultat := a + b;
            ,sortir
        si 2 : 
            resultat := a - b;
           ,sortir
        si 3 : 
            resultat := a * b;
            ,sortir
        si 4 : 
            resultat := a / b;
            ,sortir

fin

10. Algorithme avec Opérateurs Logiques
algorithme TestLogique;
variables
    a, b, c : entier;
    condition1, condition2 : booleen;
debut
    a := 5;
    b := 10;
    c := 15;
    
    condition1 := (a < b) et (b < c);
    condition2 := (a > b) ou (c > b);
    
    si condition1 et condition2 alors
        ecrire("Les deux conditions sont vraies");
    sinon
        si condition1 alors
            ecrire("Seule condition1 est vraie");
        sinon
            si condition2 alors
                ecrire("Seule condition2 est vraie");
            sinon
                ecrire("Aucune condition n'est vraie");
            fin_si;
        fin_si;
    fin_si;
fin
